        // --- 2. Interactive HTML5 Canvas Scratch Card logic ---

        function initScratchCard() {

            const canvas = document.getElementById('scratch-canvas');

            const ctx = canvas.getContext('2d');

            let isDrawing = false;


            // Paint elegant matte gold background matrix cover

            ctx.fillStyle = '#d4be9f';

            ctx.fillRect(0, 0, canvas.width, canvas.height);


            // Add fine abstract overlay texture for premium appearance

            ctx.fillStyle = 'rgba(255, 255, 255, 0.15)';

            for (let i = 0; i < 40; i++) {

                ctx.fillRect(Math.random() * canvas.width, Math.random() * canvas.height, 4, 4);

            }


            // Write call-to-action text layer on top of scratch surface

            ctx.font = '10px Montserrat';

            ctx.fillStyle = '#61533f';

            ctx.textAlign = 'center';

            ctx.textBaseline = 'middle';

            ctx.letterSpacing = '2px';

            ctx.fillText('SCRATCH WITH FINGER', canvas.width / 2, canvas.height / 2);


            // Universal event coordinate locator function

            function getCoords(e) {

                const rect = canvas.getBoundingClientRect();

                const clientX = e.touches ? e.touches[0].clientX : e.clientX;

                const clientY = e.touches ? e.touches[0].clientY : e.clientY;

                return {

                    x: clientX - rect.left,

                    y: clientY - rect.top

                };

            }


            function scratchStart(e) {

                isDrawing = true;

                scratchMove(e);

            }


            function scratchMove(e) {

                if (!isDrawing) return;

                e.preventDefault(); 

                

                const coords = getCoords(e);

                

                // Destination-out compositing breaks down painted layers natively

                ctx.globalCompositeOperation = 'destination-out';

                ctx.beginPath();

                ctx.arc(coords.x, coords.y, 24, 0, Math.PI * 2); // Comfortable pointer brush diameter

                ctx.fill();

            }


            function scratchEnd() {

                isDrawing = false;

            }


            // Bind Event Listeners natively for both desktop mouse environments and mobile touch nodes

            canvas.addEventListener('mousedown', scratchStart);

            canvas.addEventListener('mousemove', scratchMove);

            window.addEventListener('mouseup', scratchEnd);


            canvas.addEventListener('touchstart', scratchStart);

            canvas.addEventListener('touchmove', scratchMove);

            window.addEventListener('touchend', scratchEnd);

        }


        // --- 3. Production Grade Realtime Countdown Engine ---

        const targetWeddingDate = new Date('December 12, 2026 17:00:00').getTime();


        const clockInterval = setInterval(function() {

            const now = new Date().getTime();

            const distance = targetWeddingDate - now;


            const days = Math.floor(distance / (1000 * 60 * 60 * 24));

            const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));

            const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));

            const seconds = Math.floor((distance % (1000 * 60)) / 1000);


            document.getElementById('days').innerText = String(days).padStart(2, '0');

            document.getElementById('hours').innerText = String(hours).padStart(2, '0');

            document.getElementById('mins').innerText = String(minutes).padStart(2, '0');

            document.getElementById('secs').innerText = String(seconds).padStart(2, '0');


            // If the countdown finishes, clear updating functions

            if (distance < 0) {

                clearInterval(clockInterval);

                document.getElementById('days').innerText = "00";

                document.getElementById('hours').innerText = "00";

                document.getElementById('mins').innerText = "00";

                document.getElementById('secs').innerText = "00";

            }

        }, 1000);
